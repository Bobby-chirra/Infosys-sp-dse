1. First Non-Repeating Character
Given a string S, find the first character that appears only once. If no such character exists, return '#'. 
import java.util.*;
public class Solution {
    public static char firstNonRepeating(String s) {
        LinkedHashMap<Character, Integer> map = new LinkedHashMap<>();
        for (char c : s.toCharArray()) {
            map.put(c, map.getOrDefault(c, 0) + 1);
        }
        for (char c : map.keySet()) {
            if (map.get(c) == 1) return c;
        }
        return '#';
    }
}
2. Element Frequency Analysis
Given an array of integers, return a 2-element array where: [0] = number of elements that appear more than once, [1] = number of elements that appear exactly once 
import java.util.*;
public class Solution {
    public static int[] freqAnalysis(int[] arr) {
        Map<Integer, Integer> freq = new HashMap<>();
        for (int num : arr) {
            freq.put(num, freq.getOrDefault(num, 0) + 1);
        }
        int repeated = 0, unique = 0;
        for (int count : freq.values()) {
            if (count > 1) repeated++;
            else if (count == 1) unique++;
        }
        return new int[]{repeated, unique};
    }
}
3. Reverse Linked List
Standard reverse singly linked list (LeetCode 206 type) 
class ListNode {
    int val; ListNode next;
    ListNode(int x) { val = x; }
}
public class Solution {
    public ListNode reverseList(ListNode head) {
        ListNode prev = null, curr = head;
        while (curr != null) {
            ListNode next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
        }
        return prev;
    }
}
4. Summer Array Rearrange
Given an array, find minimum number of adjacent swaps required to bring all even numbers to the left half and odd numbers to the right half. 
public class Solution {
    public static int minSwaps(int[] arr) {
        int n = arr.length;
        int evenPos = 0;
        int swaps = 0;
        for (int i = 0; i < n; i++) {
            if (arr[i] % 2 == 0) {
                swaps += i - evenPos;
                evenPos++;
            }
        }
        return swaps;
    }
}

5. Maximum Monsters You Can Defeat
You are given N monsters with power[i] and bonus[i]. You start with experience E. You can defeat a monster only if your current experience ≥ power[i]. After defeating, your experience becomes E + bonus[i]. Return maximum number of monsters you can defeat (you can choose order). 
import java.util.*;
public class Solution {
    public static int maxMonsters(int[] power, int[] bonus, int E) {
        int n = power.length;
        Integer[][] monsters = new Integer[n][2];
        for (int i = 0; i < n; i++) {
            monsters[i][0] = power[i];
            monsters[i][1] = bonus[i];
        }
        Arrays.sort(monsters, (a,b) -> Integer.compare(a[0], b[0]));
        int count = 0;
        for (Integer[] m : monsters) {
            if (E >= m[0]) {
                E += m[1];
                count++;
            } else break;
        }
        return count;
    }
}
6. Maximum Subarray Sum
Classic Kadane's algorithm (max contiguous subarray sum) 
public class Solution {
    public static int maxSubArray(int[] arr) {
        int maxSoFar = arr[0], maxEndingHere = arr[0];
        for (int i = 1; i < arr.length; i++) {
            maxEndingHere = Math.max(arr[i], maxEndingHere + arr[i]);
            maxSoFar = Math.max(maxSoFar, maxEndingHere);
        }
        return maxSoFar;
    }
}

You have two operations: Cost X: Choose any subarray and decrease every element by 1. Cost Y: Choose any single element and set it to 0. Find minimum cost to make entire array zero. (Note: Needs greedy or DP solution) 
public class Solution {
    public static long minCost(int[] arr, int X, int Y) {
        long costX = 0, costY = 0;
        for (int a : arr) {
            costX += (long) a * X;
            costY += Y;

8. Count Good Arrays
Count number of arrays of length K using numbers 1 to N such that for every i, a[i+1] % a[i] == 0. Answer modulo 10000. 
public class Solution {
    public static int countArrays(int N, int K) {
        int MOD = 10000;
        int[][] dp = new int[K+1][N+1];
        for (int i = 1; i <= N; i++) dp[1][i] = 1;
        for (int len = 2; len <= K; len++) {
            for (int last = 1; last <= N; last++) {
                for (int prev = 1; prev <= last; prev++) {
                    if (last % prev == 0) {
                        dp[len][last] = (dp[len][last] + dp[len-1][prev]) % MOD;
                    }
                }
            }
        }
        int total = 0;
        for (int i = 1; i <= N; i++) total = (total + dp[K][i]) % MOD;
        return total;
    }
}

10. Maximum Vacation Days
You have N days, some days have obligations (given in array). You can cancel at most K obligations. Return maximum number of consecutive free days you can get. 
import java.util.*;
public class Solution {
    public static int maxVacation(int N, int[] obligations, int K) {
        Arrays.sort(obligations);
        List<Integer> gaps = new ArrayList<>();
        gaps.add(obligations[0] - 1);
        for (int i = 1; i < obligations.length; i++) {
            gaps.add(obligations[i] - obligations[i-1] - 1);
        }
        gaps.add(N - obligations[obligations.length-1]);
        Collections.sort(gaps, Collections.reverseOrder());
        int maxFree = 0;
        for (int i = 0; i <= K && i < gaps.size(); i++) {
            maxFree = Math.max(maxFree, gaps.get(i));
        }
        return maxFree;
    }
}
11. Smallest Base with All Same Digits
Find smallest base B (>1) such that when number M is written in base B, all digits are same. Return -1 if impossible. 
public class Solution {
    public static int smallestBase(int M) {
        for (int b = 2; b <= M; b++) {
            int temp = M;
            int digit = temp % b;
            boolean same = true;
            while (temp > 0) {
                if (temp % b != digit) { same = false; break; }
                temp /= b;
            }
            if (same) return b;
        }
        return -1;
    }
}

13. Longest Increasing Subsequence with Special Condition
Find longest subsequence where for every pair (x,y) with x before y: (x & y) * 2 < (x | y) and x < y 
import java.util.*;
public class Solution {
    public static int longestSpecialSubsequence(int[] arr) {
        int n = arr.length;
        int[] dp = new int[n];
        Arrays.fill(dp, 1);
        int max = 1;
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < i; j++) {
                if (arr[i] > arr[j] && (long)(arr[j] & arr[i]) * 2 < (arr[j] | arr[i])) {
                    dp[i] = Math.max(dp[i], dp[j] + 1);
                }
            }
            max = Math.max(max, dp[i]);
        }
        return max;
    }
}
14. Gift Box Packing
You have N gift boxes in a line, each containing a gift type. You must take exactly K consecutive boxes. Return maximum number of distinct gift types you can get. 
import java.util.*;
public class Solution {
    public static int maxValue(int[] gifts, int K) {
        int n = gifts.length;
        int maxDistinct = 0;
        for (int i = 0; i <= n-K; i++) {
            Set<Integer> set = new HashSet<>();
            for (int j = i; j < i+K; j++) {
                set.add(gifts[j]);
            }
            maxDistinct = Math.max(maxDistinct, set.size());
        }
        return maxDistinct;
    }
}
15. Maximum XOR Subset (size ≤ N/2)
Choose at most N/2 elements from array such that their XOR is maximized. 
public class Solution {
    public static int maxXorSubset(int[] arr) {
        int n = arr.length;
        int maxXor = 0;
        for (int mask = 0; mask < (1 << n); mask++) {
            if (Integer.bitCount(mask) > n/2) continue;
            int xor = 0;
            for (int i = 0; i < n; i++) {
                if ((mask & (1 << i)) != 0) xor ^= arr[i];
            }
            maxXor = Math.max(maxXor, xor);
        }
        return maxXor;
    }
}

17. Minimum Days to Make Terrain Strictly Decreasing
You have heights of mountains in a row. Each day you can reduce any mountain height by 1,2,3,…d on day d (i.e., day 1 → max reduce 1, day 2 → max reduce 3 total, etc.). Find minimum days to make the terrain strictly decreasing from left to right. 
public class Solution {
    public static int minDays(int[] height) {
        int left = 0, right = 0;
        for (int h : height) right = Math.max(right, h);
        while (left < right) {
            int mid = left + (right - left) / 2;
            if (canMake(height.clone(), mid)) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        return left;
    }
    private static boolean canMake(int[] h, int days) {
        for (int i = 1; i < h.length; i++) {
            if (h[i-1] <= h[i]) {
                int diff = h[i] - h[i-1] + 1;
                long reduce = (long)days * (days + 1) / 2;
                if (reduce < diff) return false;
            }
        }
        return true;
    }
}
